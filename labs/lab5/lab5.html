1. Define a function max() that takes two numbers as arguments and returns the largest of them. Use the if-then-else construct available in Javascript.

    
      function max(x, y) {
        if (x > y) return x;
        else return y;
      }

      // or
      let max = (x, y) => x >y?x:y

      // run
      console.log(max(10,12)); // => result: 12
    
  
    
2. Define a function maxOfThree() that takes three numbers as arguments and returns the largest of them.

    
      let maxOfThree = (x, y, z) => {
        let max = x;
        if(y> max) max = y;
        if(z> max) max = z;
        return max;
      }
      // run
      console.log(maxOfThree(3, 5, 12)); // => result: 12
    
  
    
3. Write a function isVowel() that takes a character (i.e. a string of length 1) and returns true if it is a vowel, false otherwise.

    
      function isVowel(char){
        if (char.length === 1){
          var vowels = "aeiou";
          return vowels.indexOf(char) >= 0 ? true : false;
        }
        return false;
      }
      // run
      console.log(isVowel('d')) // => false
      console.log(isVowel('e')) // => true
      console.log(isVowel('ae')) // => false
    
  
    
4. Define a function sum() and a function multiply() that sums and multiplies (respectively) all the numbers in an array of numbers. For example, sum([1,2,3,4]) should return 10, and multiply([1,2,3,4]) should return 24.

    
      var sum = (array) => array.reduce( (x, y) => x + y);

      var multiply = (array) => array.reduce( (x, y) => x * y);

      // run
      console.log(sum([1,2,3,4])) // Result = 4;
      console.log(multiply([1,2,3,4])) // Result = 24;
    
  
    
5. Define a function reverse() that computes the reversal of a string. For example, reverse("jag testar") should return the string "ratset gaj".

    
      let reverse = (s) => s.split("").reverse().join("");
      // run
      console.log(reverse("jag testar")) // Result = "ratset gaj";
    
  
    
6. Write a function findLongestWord() that takes an array of words and returns the length of the longest one.

    
      var findLongestWord = (array) => array.reduce( (x, y) => y.length>x?y.length:x, 0);

      // run
      console.log(findLongestWord(['abc','abcd', '4asssss', 'aaa5'])); // => result = 7
